/* --- Tunable settings --- */
#define PCM_SAMPLE_RATE 16000U // PCM sampling rate
#define PDM_DECIMATION PDM_FILTER_DEC_FACTOR_64
#define PCM_OUT_SAMPLES 16U                                // PDM2PCM output samples per call (per channel)
#define PDM_CHANNELS 2U                                    // mic1 (ch0) + mic2 (ch1)
#define PDM_BYTES_PER_CHANNEL ((PCM_OUT_SAMPLES * 64) / 8) // decimation=64 -> 64 PDM
bits per PCM sample
#define PDM_BYTES_PER_FRAME (PDM_BYTES_PER_CHANNEL * PDM_CHANNELS) // bytes
    consumed per PDM_Filter call
#define PDM_BUFFER_SIZE (PDM_BYTES_PER_FRAME * 2) // double buffer (DMA circular)
    /* Buffers */
    static uint8_t pdmBuf[PDM_BUFFER_SIZE];                    // DMA target (circular)
static int16_t pcmInterleaved[PCM_OUT_SAMPLES * PDM_CHANNELS]; // out of PDM_Filter
(interleaved) static int16_t pcmLeft[PCM_OUT_SAMPLES];         // chest mic
static int16_t pcmRight[PCM_OUT_SAMPLES];                      // ambient mic
/* PDM2PCM handler and config (UM2372) */
static PDM_Filter_Handler_t PDM1_FilterHandler;
static PDM_Filter_Config_t PDM1_FilterConfig;
/* --- CMSIS biquad filter structures --- */
#define HEART_BIQUAD_STAGES 2
#define LUNG_BIQUAD_STAGES 2
/* Coeffs are in CMSIS order: {b0,b1,b2, a1, a2} per stage.
 Note: CMSIS expects feedback coefficients with sign such that
 the processing uses the form with +a1*y[n-1], +a2*y[n-2]
 (see CMSIS docs). The arrays below are already prepared.
*/
/* Heart bandpass ≈ 20 - 600 Hz (fs = 16 kHz) - 4th order (2 stages) */
static float heart_biquad_coeffs[5 * HEART_BIQUAD_STAGES] = {
    /* stage 0 */ 1.0f, 2.0f, 1.0f, 1.9889416051454945f, -0.9890070054442315f,
    /* stage 1 */ 1.0f, -2.0f, 1.0f, 1.6892470409865235f, -0.7338101667589392f};
/* Lung bandpass ≈ 100 - 2000 Hz (fs = 16 kHz) - 4th order (2 stages) */
static float lung_biquad_coeffs[5 * LUNG_BIQUAD_STAGES] = {
    /* stage 0 */ 1.0f, 2.0f, 1.0f, 1.9449249126409676f, -0.9465723162891281f,
    /* stage 1 */ 1.0f, -2.0f, 1.0f, 1.0478740914248466f, -0.3902948163334587f};
/* state buffers for filters */
static float heart_biquad_state[2 * HEART_BIQUAD_STAGES];
static float lung_biquad_state[2 * LUNG_BIQUAD_STAGES];
static arm_biquad_casd_df1_inst_f32 heartFilter;
static arm_biquad_casd_df1_inst_f32 lungFilter;
/* Forward declarations (CubeMX provides SAI handle and MX init functions) */
extern SAI_HandleTypeDef hsai_BlockA1; // generated by CubeMX
void Error_Handler(void);
/* Simple detection-state for demo */
static float heart_env_prev = 0.0f;
static uint32_t last_heart_ms = 0;
static const float HEART_DETECT_THRESHOLD = 0.02f; // tune this
static const uint32_t HEART_MIN_GAP_MS = 200;      // minimum gap between beats (tune)
/* Utility: convert int16_t PCM -> float and normalize to [-1,1] */
static void pcm16_to_float(const int16_t *in, float *out, uint32_t n)
{
    for (uint32_t i = 0; i < n; i++)
        out[i] = ((float)in[i]) / 32768.0f;
}
/* Very simple envelope: absolute + single-pole lowpass */
static void compute_envelope(const float *x, uint32_t n, float *env_out, float *env_state)
{
    const float alpha = 0.05f; // envelope smoothing (tune)
    float env = *env_state;
    for (uint32_t i = 0; i < n; i++)
    {
        float a = fabsf(x[i]);
        env = alpha * a + (1.0f - alpha) * env;
    }
    *env_state = env;
    *env_out = env;
}
/* Process a block of PCM samples (PCM_OUT_SAMPLES) for each channel.
 Called from DMA callback when a chunk is ready.
*/
static void process_pcm_block(int16_t *pcmL, int16_t *pcmR, uint32_t N)
{
    float bufL[PCM_OUT_SAMPLES];
    float bufR[PCM_OUT_SAMPLES];
    pcm16_to_float(pcmL, bufL, N);
    pcm16_to_float(pcmR, bufR, N);
    /* Apply filters (use CMSIS DF1 biquad cascade) */
    arm_biquad_cascade_df1_f32(&heartFilter, bufL, bufL, N); // in-place
    arm_biquad_cascade_df1_f32(&lungFilter, bufR, bufR, N);  // NOTE: in demo we filter ambient on
    lungFilter just as example
        /* Compute envelope / simple energy */
        float envL = 0,
              envR = 0;
    compute_envelope(bufL, N, &envL, &heart_env_prev);
    compute_envelope(bufR, N, &envR, &heart_env_prev); // reuse state for demo; you can have
    separate states
        /* Example heartbeat detection (very simple): detect rising envelope above threshold */
        if (envL > HEART_DETECT_THRESHOLD)
    {

        if ((now_ms - last_heart_ms) > HEART_MIN_GAP_MS)

    }
}
/* Called from HAL when DMA half-transfer completes */
void HAL_SAI_RxHalfCpltCallback(SAI_HandleTypeDef *hsai)
{
    if (hsai == &hsai_BlockA1)
    {
        /* process first half of pdmBuf */
        uint8_t *p = &pdmBuf[0];
        /* call PDM_Filter for this chunk */
        /* PDM_Filter expects byte-packed PDM stream; returns interleaved int16 PCM in
       pcmInterleaved */
        PDM_Filter(p, pcmInterleaved, &PDM1_FilterHandler);
        /* For stereo, use deInterleave helper */
        PDM_Filter_deInterleave((uint16_t *)pcmInterleaved, (uint16_t *)pcmLeft, (uint16_t *)pcmRight,
                                PCM_OUT_SAMPLES);
        process_pcm_block(pcmLeft, pcmRight, PCM_OUT_SAMPLES);
    }
}
/* Called from HAL when DMA transfer complete (second half) */
void HAL_SAI_RxCpltCallback(SAI_HandleTypeDef *hsai)
{
    if (hsai == &hsai_BlockA1)
    {
        uint8_t *p = &pdmBuf[PDM_BUFFER_SIZE / 2];
        PDM_Filter(p, pcmInterleaved, &PDM1_FilterHandler);
        PDM_Filter_deInterleave((uint16_t *)pcmInterleaved, (uint16_t *)pcmLeft, (uint16_t *)pcmRight,
                                PCM_OUT_SAMPLES);
        process_pcm_block(pcmLeft, pcmRight, PCM_OUT_SAMPLES);
    }
}
int main(void)
{
    HAL_Init();
    SystemClock_Config();
    MX_GPIO_Init();
    MX_DMA_Init();
    MX_CRC_Init();
    MX_SAI1_Init(); // CubeMX generated (SAI1 block A, master rx, 8-bit data, DMA circular)
    /* init CMSIS biquad instances */
    arm_biquad_cascade_df1_init_f32(&heartFilter, HEART_BIQUAD_STAGES, heart_biquad_coeffs,
                                    heart_biquad_state);
    arm_biquad_cascade_df1_init_f32(&lungFilter, LUNG_BIQUAD_STAGES, lung_biquad_coeffs,
                                    lung_biquad_state);
    /* PDM2PCM library setup (UM2372) */
    __HAL_RCC_CRC_CLK_ENABLE(); // required by the PDM2PCM library
    /* initialize handler */
    PDM1_FilterHandler.bit_order = PDM_FILTER_BIT_ORDER_LSB;
    PDM1_FilterHandler.endianness = PDM_FILTER_ENDIANNESS_BE;
    PDM1_FilterHandler.high_pass_tap = 2122358088u;               // example Q31 coefficient (~0.98). Tune or 0
    to bypass.PDM1_FilterHandler.out_ptr_channels = PDM_CHANNELS; // output channels
    PDM1_FilterHandler.in_ptr_channels = PDM_CHANNELS;            // input channels (stereo)
    PDM_Filter_Init(&PDM1_FilterHandler);
    /* set config */
    PDM1_FilterConfig.output_samples_number = PCM_OUT_SAMPLES;
    PDM1_FilterConfig.mic_gain = 24;                                // in dB ([-12..+51] typical). Tune gain to avoid clipping.
    PDM1_FilterConfig.decimation_factor = PDM_FILTER_DEC_FACTOR_64; // choose according to
    PDM clock
        PDM_Filter_setConfig(&PDM1_FilterHandler, &PDM1_FilterConfig);
    /* start SAI DMA reception into pdmBuf */
    if (HAL_SAI_Receive_DMA(&hsai_BlockA1, (uint8_t *)pdmBuf, PDM_BUFFER_SIZE) != HAL_OK)
    {
        Error_Handler();
    }
    /* main loop - actual processing happens in DMA callbacks */
    while (1)
    {
        HAL_Delay(1000);
        // optionally send status over UART, etc.
    }
}
